 ORG 3000H
SWAP: PUSH BX   ;APILO REGISTROS
 PUSH CX
 PUSH DX
 MOV BX, SP     ;PONGO EN BX LA DIR DEL TOPE DE PILA
 ADD BX, 8      ;BX APUNTA A LA DIR DONDE ESTA LA DIR DEL D2
                ;2 POR CADA REGISTRO (2*3) + 2 POR EL CALL SWAP
                
 MOV AX, [BX]   ;PONGO EN AX LA DIR DEL D2
 ADD BX, 2      ;BX APUNTA A LA DIR DONDE ESTA LA DIR DEL D1
 MOV CX, [BX]   ;PONGO EN CX LA DIR DEL D1
 MOV BX, CX     ;PONGO EN BX LA DIR DEL D1
 MOV DX, [BX]   ;PONGO EN DX EL D1
 MOV BX, AX     ;PONGO EN BX LA DIR DEL D2
 MOV AX, [BX]   ;PONGO EN AX EL D2
 MOV [BX], DX   ;PONGO EN LA DIRECCION DEL D2, EL D1
 MOV BX, CX     ;PONGO EN BX LA DIR DEL D1
 MOV [BX], AX   ;PONGO EN LA DIRECCION DEL D1, EL D2
 
 POP DX
 POP CX
 POP BX
 RET

 ORG 2000H
 PUSH AX            ;APILO AX
 MOV AX, OFFSET D1  ;APILO PARAMETROS
 PUSH AX
 MOV AX, OFFSET D2
 PUSH AX
 CALL SWAP
 POP AX             ;DESAPILO PARAMETROS
 POP AX
 POP AX		  ;DESAPILO AX
 HLT
 END
