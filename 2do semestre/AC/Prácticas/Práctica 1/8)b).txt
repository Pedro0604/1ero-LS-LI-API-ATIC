  ORG 1000H
  CADENA DB "HoLaZ"
  CERO DB 0
  A_MIN DB "a"
  Z_MIN DB "z"
  CANT_MIN DW ?

  ORG 3000H
CONTAR_MIN: MOV CX, 0
LOOP:MOV AH, [BX]
  CMP AH, 0
  JZ FIN
  CMP AH, A_MIN
  JZ ES_MIN
  JNS MAY_A
  JMP NO_ES_MIN

MAY_A: CMP AH, Z_MIN
  JZ ES_MIN
  JS ES_MIN
  JMP NO_ES_MIN

ES_MIN: INC CX
NO_ES_MIN: INC BX
  JMP LOOP
FIN:RET

  ORG 2000H
  MOV  BX, OFFSET CADENA
  CALL CONTAR_MIN
  MOV CANT_MIN, CX
  HLT
  END
