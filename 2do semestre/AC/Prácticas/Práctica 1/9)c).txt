  ORG 1000H
  BY DB 00000001B
  N DW 2
  
  ORG 3000H
ROTAR_IZQ: ADD AH, AH
  JC CARRY
  JMP FIN_ROT
CARRY: INC AH
FIN_ROT: MOV [BX], AH
  RET
  
ROTAR_IZQN: CMP CX, 0
  JZ FIN_ROTN
  CALL ROTAR_IZQ
  DEC CX
  JMP ROTAR_IZQN
FIN_ROTN: RET
  
ROTAR_DER: MOV CX, 8
  SUB CX, DX
  CALL ROTAR_IZQN
  RET
  
  ORG 2000H
  MOV DX, N
  MOV AH, BY
  MOV BX, OFFSET BY
  CALL ROTAR_DER
  HLT
  END
