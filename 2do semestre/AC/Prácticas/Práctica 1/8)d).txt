  ORG 1000H
  VOCALES DB "aeiouAEIOU"
  CADENA DB "HOLA me llamo Pedro."
  CERO DB 0
  CANT_VOC DW ?

  ORG 3000H
ES_VOCAL: MOV BX, OFFSET VOCALES
  MOV CX, OFFSET CADENA
  MOV AL,0
LOOP: CMP CX, BX
  JZ FIN
  CMP AH, [BX]
  JZ ES_V
  INC BX
  JMP LOOP
ES_V: MOV AL, 0FFH
RET

CONTAR_VOC: MOV CX, 0
LAZO: MOV AH, [BX]
  CMP AH, 0
  JZ FIN
  PUSH BX
  PUSH CX
  CALL ES_VOCAL
  POP CX
  POP BX
  CMP AL, 0FFH
  JNZ NO_ES
  INC CX
NO_ES: INC BX
  JMP LAZO
FIN:RET

  ORG 2000H
  MOV  BX, OFFSET CADENA
  CALL CONTAR_VOC
  MOV CANT_VOC, CX
  HLT
  END
