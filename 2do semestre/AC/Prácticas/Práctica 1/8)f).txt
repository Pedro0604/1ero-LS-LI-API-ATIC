  ORG 1000H
  CADENA DB "MI MOTO ALPINA DERRAPANTE"
  CERO DB 0
  OG DB "A"
  RMP DB "E"
  
  ORG 3000H
REEMPLAZAR_CAR: MOV BX, SP
  ADD BX, 2
  MOV CL, [BX]
  ADD BX, 2
  MOV AL, [BX]
  ADD BX, 2
  MOV BX, [BX]
  
LOOP: MOV AH, [BX]
  CMP AH, 0
  JZ FIN
  CMP AH, AL
  JNZ NO_ES
  MOV [BX], CL
NO_ES: INC BX
  JMP LOOP
  
FIN: RET  
  
  ORG 2000H
  MOV AX, OFFSET CADENA
  PUSH AX
  MOV AL, OG
  PUSH AX
  MOV AL, RMP
  PUSH AX
  CALL REEMPLAZAR_CAR
  POP AX
  POP AX
  POP AX
  HLT
  END
