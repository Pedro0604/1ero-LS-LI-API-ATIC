  ORG 1000H
  VOCALES DB "aeiouAEIOU"
  CARACTER DB "R"
  RES DB ?

  ORG 3000H
ES_VOCAL: MOV BX, OFFSET VOCALES
  MOV CX, OFFSET CARACTER
  MOV AL,0
LOOP: CMP CX, BX
  JZ FIN
  CMP AH, [BX]
  JZ ES_V
  INC BX
  JMP LOOP
ES_V: MOV AL, 0FFH
FIN:RET

  ORG 2000H
  MOV  AH, CARACTER
  CALL ES_VOCAL
  MOV RES, AL
  HLT
  END
