ORG 1000H
MSJ DB "INGRESE UN NUMERO: "
NUM DB ?
CERO DB "CERO"
UNO DB "UNO"
DOS DB "DOS"
TRES DB "TRES"
CUATRO DB "CUATRO"
CINCO DB "CINCO"
SEIS DB "SEIS"
SIETE DB "SIETE"
OCHO DB "OCHO"
NUEVE DB "NUEVE"
NO_ES DB "NO SE INGRESO UN NUMERO."
SALTO DB 0AH
F DB "FIN."



ORG 3000H
INGRESE_UN_NUMERO: MOV BX, OFFSET MSJ
 MOV AL, OFFSET NUM-OFFSET MSJ
 INT 7
 MOV BX, OFFSET NUM
 INT 6
 RET
 
 
SALTO_DE_LINEA: MOV BX, OFFSET SALTO
 MOV AL, OFFSET F-OFFSET SALTO
 INT 7
 RET
 
 
IMPRIME_NUMERO: CMP BYTE PTR [BX], 30H
 JZ CE
 CMP BYTE PTR [BX], 31H
 JZ UN
 CMP BYTE PTR [BX], 32H
 JZ D
 CMP BYTE PTR [BX], 33H
 JZ TR
 CMP BYTE PTR [BX], 34H
 JZ CUA
 CMP BYTE PTR [BX], 35H
 JZ CIN
 CMP BYTE PTR [BX], 36H
 JZ SE
 CMP BYTE PTR [BX], 37H
 JZ SIE
 CMP BYTE PTR [BX], 38H
 JZ OCH
 CMP BYTE PTR [BX], 39H
 JZ NUE
 
 
 PUSH BX
 MOV AH, 0
 MOV BX, OFFSET NO_ES
 MOV AL, OFFSET SALTO-OFFSET NO_ES
 INT 7
 CALL SALTO_DE_LINEA
 JMP FIN
 
 CE: PUSH BX
  CMP AH, 0
  JNZ DOS_CEROS
  INC AH
  MOV BX, OFFSET CERO
  MOV AL, OFFSET UNO-OFFSET CERO
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
 
 UN: PUSH BX
  MOV AH, 0
  MOV BX, OFFSET UNO
  MOV AL, OFFSET DOS-OFFSET UNO
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
 
 D: PUSH BX
  MOV AH, 0
  MOV BX, OFFSET DOS
  MOV AL, OFFSET TRES-OFFSET DOS
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
  
 TR: PUSH BX
  MOV AH, 0
  MOV BX, OFFSET TRES
  MOV AL, OFFSET CUATRO-OFFSET TRES
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
  
 CUA: POP BX
  MOV AH, 0
  MOV BX, OFFSET CUATRO
  MOV AL, OFFSET CINCO-OFFSET CUATRO
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
  
 CIN: PUSH BX
  MOV AH, 0
  MOV BX, OFFSET CINCO
  MOV AL, OFFSET SEIS-OFFSET CINCO
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
  
 SE: PUSH BX
  MOV AH, 0
  MOV BX, OFFSET SEIS
  MOV AL, OFFSET SIETE-OFFSET SEIS
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
  
 SIE: PUSH BX
  MOV AH, 0
  MOV BX, OFFSET SIETE
  MOV AL, OFFSET OCHO-OFFSET SIETE
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
  
 OCH: PUSH BX
  MOV AH, 0
  MOV BX, OFFSET OCHO
  MOV AL, OFFSET NUEVE-OFFSET OCHO
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
 
 NUE: PUSH BX
  MOV AH, 0
  MOV BX, OFFSET NUEVE
  MOV AL, OFFSET NO_ES-OFFSET NUEVE
  INT 7
  CALL SALTO_DE_LINEA
  JMP FIN
  
 
DOS_CEROS: INC AH
 MOV BX, OFFSET F
 MOV AL, 4
 INT 7
FIN: POP BX
 RET
 
 
NUMEROS: MOV AH, 0
LOOP: CALL INGRESE_UN_NUMERO
 CALL IMPRIME_NUMERO
 CMP AH, 2
 JZ TERMINO
 JMP LOOP
TERMINO: RET
 
 
 
ORG 2000H
 CALL NUMEROS
 HLT
END
