ORG 1000H
MSJ DB "INGRESE UN NUMERO: "
MSJ2 DB "CARACTER NO VALIDO"
NUM DB ?


ORG 3000H
ES_NUM: MOV CL, [BX]
 CMP CL, 30H
 JNS MAYOR_QUE_0
 JMP NO_ES
MAYOR_QUE_0: CMP CL, 40H
 JS MENOR_QUE_9
NO_ES: MOV DX, 0H
 JMP FIN
MENOR_QUE_9: MOV DX, 0FFH
FIN: RET


ORG 2000H
 MOV BX, OFFSET MSJ
 MOV AL, OFFSET MSJ2-OFFSET MSJ
 INT 7
 MOV BX, OFFSET NUM
 INT 6
 MOV AL, 1
 INT 7
 CALL ES_NUM
 MOV NUM, 0AH
 MOV BX, OFFSET NUM
 MOV AL, 1
 INT 7
 CMP DX, 0H
 JNZ ES_UN_NUMERO
 MOV BX, OFFSET MSJ2
 MOV AL, OFFSET NUM-OFFSET MSJ2
 INT 7
ES_UN_NUMERO: HLT
END
