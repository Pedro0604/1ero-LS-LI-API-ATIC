ORG 1000H
 STRING DB "Organización y la Computación"
 FIN DB 0FFH
 A DB "a"
 C DB "c"
 ANT_IGUAL_A DB 0
 CANT DB 0
ORG 2000H
 MOV BX, OFFSET STRING
 MOV CX, OFFSET FIN - OFFSET STRING
 MOV AL, A
 MOV AH, C
 MOV DL, ANT_IGUAL_A
 NOTERMINO: CMP DL, 1
  JNZ ANT_NO_ES_A
   CMP [BX], AH
   JNZ ANT_NO_ES_A
    INC CANT
    MOV DL, 0
   JMP ETIQ
  ANT_NO_ES_A: CMP [BX], AL
   JNZ NO_ES_A
    MOV DL, 1
   JMP ETIQ
   NO_ES_A: MOV DL, 0
  ETIQ: INC BX
  DEC CX
  JNZ NOTERMINO
HLT
END
